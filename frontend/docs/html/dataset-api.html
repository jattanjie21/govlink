# GET facilities in West Coast Region offering antenatal care
curl https://api.govlink.gm/v1/health/facilities \
  -G \
  -H "Authorization: Bearer $GOVLINK_KEY" \
  -H "Accept: application/json" \
  -d "region=west-coast" \
  -d "facility_type=district_hospital" \
  -d "services=anc" \
  -d "per_page=50"
import os
from govlink import Client

gl = Client(api_key=os.environ["GOVLINK_KEY"])

facilities = gl.health.facilities(
    region="west-coast",
    facility_type="district_hospital",
    services=["anc"],
    per_page=50,
)

for f in facilities:
    print(f.name, f.district, f.beds_total)
import { GovLink } from "@govlink/sdk";

const gl = new GovLink({ apiKey: process.env.GOVLINK_KEY });

const { data, meta } = await gl.health.facilities({
  region: "west-coast",
  facility_type: "district_hospital",
  services: ["anc"],
  per_page: 50,
});

console.log(`${meta.total} facilities`);
